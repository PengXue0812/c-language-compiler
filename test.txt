//#include<stdio.h>


int main() {
    int a = 5;
    int sum = -1;

    // outer loop
    for (int i = 0; i < a * 2; i=i+1) {
        // inner loop
        for (int j = 0; j < a ; j=j+1) {
                int k = i * j;
                sum = sum + k;
        }
    }

    // print the value of `sum`
    print(sum);

    return 0;
}
